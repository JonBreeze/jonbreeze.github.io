m = sp.Function('m')
from sympy.abc import t, a, b, c
f_ = sp.diff(m(t), t) - (a-b)*m(t) - c
f_
Out[16]:
$$- c - \left(a - b\right) m{\left (t \right )} + \frac{d}{d t} m{\left (t \right )}$$
In [17]:

sol
Out[17]:
